# test service protocol-number rule

config system interface
    edit "zone1"
        set vdom "root"
        set ip 192.168.0.0 255.255.255.0
    next
    edit "zone2"

end

config firewall service custom
edit test

end

set srcintf zone1
set dstintf zone2
set action deny
set service test
next
end
