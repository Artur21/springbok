# test addrgrp member rule

config system interface
    edit "zone1"
        set vdom "root"
        set ip 192.168.0.0 255.255.255.0
    next
    edit "zone2"
        set vdom "root"
        set ip 192.168.1.0 255.255.255.0
    next
end


config firewall address
edit test1
set type wildcard
set wildcard 1.1.1.1 255.0.0.0
end

config firewall address
edit test2
set type ipmask
set subnet 2.2.2.2/8
end

config firewall addrgrp
edit test
set member "test1" test2
end


config firewall policy
edit 0
set srcintf zone1
set dstintf zone2
set action deny
set dstaddr test
next
end
